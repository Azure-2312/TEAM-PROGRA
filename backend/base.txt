from flask import Flask, jsonify
from flask_cors import CORS
import sqlite3
import os
from werkzeug.security import generate_password_hash

app = Flask(__name__)
CORS(app)

DATABASE = "database.db"


def conectar_db():
    return sqlite3.connect(DATABASE)


def inicializar_bd():
    conn = conectar_db()
    cursor = conn.cursor()

    cursor.execute("DROP TABLE IF EXISTS analisis")
    cursor.execute("DROP TABLE IF EXISTS clasificaciones")
    cursor.execute("DROP TABLE IF EXISTS usuarios")

    cursor.execute("""
        CREATE TABLE usuarios (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            nombre_completo TEXT NOT NULL,
            correo TEXT UNIQUE NOT NULL,
            username TEXT UNIQUE NOT NULL,
            password TEXT NOT NULL,
            foto_perfil_url TEXT,
            fecha_registro DATETIME DEFAULT CURRENT_TIMESTAMP
        )
    """)

    cursor.execute("""
        CREATE TABLE clasificaciones (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            codigo TEXT UNIQUE NOT NULL,
            nombre TEXT NOT NULL,
            recomendacion TEXT NOT NULL
        )
    """)

    cursor.execute("""
        CREATE TABLE analisis (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            usuario_id INTEGER NOT NULL,
            clasificacion_id INTEGER NOT NULL,
            nombre_analisis TEXT NOT NULL,
            imagen_url TEXT NOT NULL,
            public_id_cloudinary TEXT,
            confianza REAL NOT NULL,
            fecha DATETIME DEFAULT CURRENT_TIMESTAMP,
            FOREIGN KEY (usuario_id) REFERENCES usuarios(id),
            FOREIGN KEY (clasificacion_id) REFERENCES clasificaciones(id)
        )
    """)

    clasificaciones = [
        (
            "cacao_barrenador",
            "Cacao con barrenador",
            "Se recomienda revisar la mazorca afectada y separar los frutos con signos de perforación o daño interno."
        ),
        (
            "cacao_mazorca_negra",
            "Cacao con mazorca negra",
            "Se recomienda retirar las mazorcas afectadas y evitar el contacto con frutos sanos para reducir la propagación."
        ),
        (
            "cacao_mirido",
            "Cacao con mírido",
            "Se recomienda inspeccionar la planta y controlar la presencia de insectos que puedan dañar la superficie del cacao."
        ),
        (
            "cacao_moniliasis",
            "Cacao con moniliasis",
            "Se recomienda eliminar frutos infectados y mejorar la ventilación del cultivo para disminuir la humedad."
        ),
        (
            "cacao_sano",
            "Cacao sano",
            "Producto apto visualmente. El cacao se encuentra en buen estado."
        )
    ]

    cursor.executemany("""
        INSERT INTO clasificaciones (codigo, nombre, recomendacion)
        VALUES (?, ?, ?)
    """, clasificaciones)

    password_demo = generate_password_hash("123456")

    cursor.execute("""
        INSERT INTO usuarios (
            nombre_completo,
            correo,
            username,
            password,
            foto_perfil_url
        )
        VALUES (?, ?, ?, ?, ?)
    """, (
        "Piero Montalvo",
        "pieromontalvof@gmail.com",
        "Piermont",
        password_demo,
        ""
    ))

    conn.commit()
    conn.close()

    print("Base de datos reiniciada correctamente")


@app.route("/")
def home():
    return jsonify({
        "mensaje": "Backend Cacao funcionando correctamente"
    })


@app.route("/api/clasificaciones")
def obtener_clasificaciones():
    conn = conectar_db()
    cursor = conn.cursor()

    cursor.execute("""
        SELECT id, codigo, nombre, recomendacion
        FROM clasificaciones
    """)

    datos = cursor.fetchall()
    conn.close()

    return jsonify([
        {
            "id": d[0],
            "codigo": d[1],
            "nombre": d[2],
            "recomendacion": d[3]
        }
        for d in datos
    ])


@app.route("/api/usuarios")
def obtener_usuarios():
    conn = conectar_db()
    cursor = conn.cursor()

    cursor.execute("""
        SELECT id, nombre_completo, correo, username, foto_perfil_url, fecha_registro
        FROM usuarios
    """)

    datos = cursor.fetchall()
    conn.close()

    return jsonify([
        {
            "id": d[0],
            "nombre_completo": d[1],
            "correo": d[2],
            "username": d[3],
            "foto_perfil_url": d[4],
            "fecha_registro": d[5]
        }
        for d in datos
    ])


if __name__ == "__main__":
    inicializar_bd()
    app.run(debug=True)
